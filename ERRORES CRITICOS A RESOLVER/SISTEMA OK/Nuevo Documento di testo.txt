```python
                result['status'] = 'ERROR'

            tables = [row[0] for row in cursor.fetchall()]
            result['details']['tables'] = tables

            # Verificar conteo de canales
            if 'channels' in tables:

                channel_count = cursor.fetchone()[0]
                result['details']['channels_count'] = channel_count
                
            # Verificar conteo de videos
            if 'videos' in tables:
                cursor.execute("SELECT COUNT(*) FROM videos")
                video_count = cursor.fetchone()[0]
                result['details']['videos_count'] = video_count
                
            # Verificar integridad
            cursor.execute("PRAGMA integrity_check")
            integrity = cursor.fetchone()[0]
            result['details']['integrity'] = integrity
            
            if integrity != 'ok':
                result['status'] = 'WARNING'
                result['message'] = 'Problemas de integridad en la base de datos'
                
            conn.close()
            result['message'] = f'Base de datos OK - {len(tables)} tablas'

            result['status'] = 'ERROR'
            result['message'] = f'Error: {str(e)}'

            # Contar archivos

            json_files = list(transcriptions_dir.glob("*.json"))
            
            total_size = sum(f.stat().st_size for f in transcriptions_dir.glob("*"))
            
            result['details']['vtt_count'] = len(vtt_files)
            result['details']['json_count'] = len(json_files)
            result['details']['total_size_mb'] = f'{total_size / (1024**2):.1f} MB'
            
            if len(vtt_files) == 0:
                result['message'] = 'Sin transcripciones descargadas'
            else:
                result['message'] = f'{len(vtt_files)} transcripciones ({total_size / (1024**2):.1f} MB)'
                
        except Exception as e:
            result['status'] = 'WARNING'
            result['message'] = f'Error: {str(e)}'
            
        return result

        report = []
        report.append("=" * 60)
        report.append("REPORTE DE ESTADO DEL SISTEMA - KDP MASTER SUITE")
        report.append("=" * 60)
        report.append(f"Fecha: {results['timestamp']}")
        report.append(f"Estado General: {results['overall_status']}")
        report.append("")
        
        for comp_name, comp_data in results['components'].items():
            status_icon = {
                'OK': '✅',
                'WARNING': '⚠️',
                'ERROR': '❌'
            }.get(comp_data['status'], '❓')
            
            report.append(f"{status_icon} {comp_data['name']}: {comp_data['message']}")
            
        if results['warnings']:
            report.append("")
            report.append("⚠️ ADVERTENCIAS:")
            for warning in results['warnings']:
                report.append(f"  - {warning}")
                
        if results['errors']:
            report.append("")
            report.append("❌ ERRORES:")
            for error in results['errors']:
                report.append(f"  - {error}")
                
        if results['recommendations']:
            report.append("")
            report.append("💡 RECOMENDACIONES:")
            for rec in results['recommendations']:
                report.append(f"  - {rec}")
                
        report.append("")
        report.append("=" * 60)
        
        return "\n".join(report)
```

```python
            self.after(0, lambda: self.update_status_label(f"● Error: {str(e)}", "red"))
            
    def update_ui_with_results(self, results):
        """Actualiza la interfaz con los resultados"""
        # Actualizar estado general
        status_colors = {
            'OK': 'green',
            'WARNING': 'orange',
            'ERROR': 'red'
        }
        
        status_icons = {
            'OK': '✅',
            'WARNING': '⚠️',
            'ERROR': '❌'
        }
        
        overall_status = results['overall_status']
        self.update_status_label(
            f"{status_icons[overall_status]} SISTEMA {overall_status}",
            status_colors[overall_status]
        )
        
        self.timestamp_label.configure(
            text=f"Última verificación: {datetime.fromisoformat(results['timestamp']).strftime('%Y-%m-%d %H:%M:%S')}"
        )
        
        # Actualizar indicadores
        component_keys = {
            'database': 'database',
            'directories': 'directories',
            'files': 'files',
            'python': 'python',
            'disk_space': 'disk',
            'memory': 'memory',
            'network': 'network',
            'youtube_api': 'youtube'
        }
        
        for comp_key, indicator_key in component_keys.items():
            if comp_key in results['components'] and indicator_key in self.indicators:
                comp_data = results['components'][comp_key]
                indicator = self.indicators[indicator_key]
                
                icon = status_icons.get(comp_data['status'], '❓')
                indicator['icon'].configure(text=icon)
                indicator['status'].configure(
                    text=comp_data['message'],
                    text_color=status_colors.get(comp_data['status'], 'gray')
                )
                
                # Cambiar color del frame según estado
                frame_colors = {
                    'OK': '#2d5016',
                    'WARNING': '#7d5a00',
                    'ERROR': '#7d1a1a'
                }
                indicator['frame'].configure(fg_color=frame_colors.get(comp_data['status'], '#333333'))
                
        # Actualizar treeview de componentes
        for item in self.components_tree.get_children():
            self.components_tree.delete(item)
            
        for comp_name, comp_data in results['components'].items():
            self.components_tree.insert("", "end", values=(
                comp_data['name'],
                comp_data['status'],
                comp_data['message']
            ))
            
        # Actualizar logs
        self.logs_text.configure(state="normal")
        self.logs_text.delete("1.0", "end")
        self.logs_text.insert("1.0", self.system_checker.get_system_report())
        self.logs_text.configure(state="disabled")
        
        # Mostrar notificación si hay errores
        if overall_status == 'ERROR':
            messagebox.showwarning(
                "Errores Detectados",
                "Se han detectado errores en el sistema. Revise el tab de Componentes para más detalles."
            )
            
    def update_status_label(self, text, color):
        """Actualiza la etiqueta de estado"""
        self.status_label.configure(text=text, text_color=color)
        
    def toggle_auto_refresh(self):
        """Activa/desactiva el auto-refresh"""
        self.auto_refresh_enabled = not self.auto_refresh_enabled
        
        if self.auto_refresh_enabled:
            self.btn_auto_refresh.configure(
                text=f"⏱️ Auto-Refresh: ON ({self.auto_refresh_interval}s)",
                fg_color="green"
            )
            self.auto_refresh_loop()
        else:
            self.btn_auto_refresh.configure(
                text="⏱️ Auto-Refresh: OFF",
                fg_color="gray"
            )
            
    def auto_refresh_loop(self):
        """Bucle de auto-refresh"""
        if self.auto_refresh_enabled:
            threading.Thread(target=self.run_system_check).start()
            self.after(self.auto_refresh_interval * 1000, self.auto_refresh_loop)
            
    def on_component_double_click(self, event):
        """Maneja doble clic en un componente"""
        selection = self.components_tree.selection()
        if not selection:
            return
            
        item = self.components_tree.item(selection[0])
        component_name = item['values'][0]
        
        if self.current_results:
            for comp_key, comp_data in self.current_results['components'].items():
                if comp_data['name'] == component_name:
                    details = json.dumps(comp_data.get('details', {}), indent=2, ensure_ascii=False)
                    messagebox.showinfo(
                        f"Detalles: {component_name}",
                        f"{comp_data['message']}\n\nDetalles:\n{details}"
                    )
                    break
                    
    def export_report(self):
        """Exporta el reporte a archivo"""
        if not self.current_results:
            messagebox.showwarning("Advertencia", "No hay resultados para exportar")
            return
            
        from tkinter import filedialog
        
        file_path = filedialog.asksaveasfilename(
            title="Exportar Reporte",
            defaultextension=".txt",
            filetypes=[("Archivos de texto", "*.txt"), ("JSON", "*.json"), ("Todos los archivos", "*.*")]
        )
        
        if file_path:
            try:
                if file_path.endswith('.json'):
                    with open(file_path, 'w', encoding='utf-8') as f:
                        json.dump(self.current_results, f, indent=2, ensure_ascii=False)
                else:
                    with open(file_path, 'w', encoding='utf-8') as f:
                        f.write(self.system_checker.get_system_report())
                        
                messagebox.showinfo("Éxito", f"Reporte exportado a:\n{file_path}")
                
            except Exception as e:
                messagebox.showerror("Error", f"No se pudo exportar el reporte:\n{str(e)}")
```

```python
            )

                text="❌ SISTEMA ERROR",
                fg_color="red",
                hover_color="darkred"
            )

        """Verificación periódica cada 5 minutos"""
        results = self.system_checker.check_all()
        
        # Actualizar botón

            self.system_ok_button.configure(

                fg_color="red"
            )

        self.after(300000, self.periodic_system_check)
```
