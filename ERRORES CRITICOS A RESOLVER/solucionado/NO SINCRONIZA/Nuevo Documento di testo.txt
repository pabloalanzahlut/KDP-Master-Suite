def get_dashboard_stats():
    """Obtiene estadísticas reales de la BD local"""
    try:
        if not DB_PATH.exists():
            return {'error': f'Base de datos no encontrada: {DB_PATH}'}, 500
            
        conn = sqlite3.connect(str(DB_PATH))
        conn.row_factory = sqlite3.Row  # Permite acceso por nombre de columna
        cursor = conn.cursor()
        
        # Canales activos
        cursor.execute("SELECT COUNT(*) FROM channels WHERE active = 1")
        canales_activos = cursor.fetchone()[0]
        
        # Videos totales detectados
        cursor.execute("SELECT COUNT(*) FROM videos")
        videos_totales = cursor.fetchone()[0]
        
        # Videos NUEVOS (no descargados aún)
        cursor.execute("""
            SELECT COUNT(*) FROM videos 
            WHERE downloaded = 0 OR downloaded IS NULL
        """)
        videos_pendientes = cursor.fetchone()[0]
        
        # Videos en procesamiento
        cursor.execute("SELECT COUNT(*) FROM videos WHERE status = 'processing'")
        videos_procesando = cursor.fetchone()[0]
        
        # Videos completados
        cursor.execute("""
            SELECT COUNT(*) FROM videos 
            WHERE status = 'completed' OR downloaded = 1
        """)
        videos_completados = cursor.fetchone()[0]
        
        # Videos fallidos
        cursor.execute("SELECT COUNT(*) FROM videos WHERE status = 'failed'")
        videos_fallidos = cursor.fetchone()[0]
        
        # Actividad reciente (últimos 5 videos)
        cursor.execute("""
            SELECT title, channel_name, created_at, status 
            FROM videos 
            ORDER BY created_at DESC 
            LIMIT 5
        """)
        actividad_reciente = [
            {
                'title': row['title'] or 'Sin título',
                'channel': row['channel_name'] or 'Desconocido',
                'date': row['created_at'],
                'status': row['status']
            }
            for row in cursor.fetchall()
        ]
        
        conn.close()
        
        return {
            'canales_activos': canales_activos,
            'videos_totales': videos_totales,
            'pendientes': videos_pendientes,
            'procesando': videos_procesando,
            'completados': videos_completados,
            'fallidos': videos_fallidos,
            'actividad_reciente': actividad_reciente,
            'ultima_actualizacion': datetime.now().strftime('%H:%M:%S'),
            'db_path': str(DB_PATH)
        }, 200
        
    except sqlite3.Error as e:
        return {'error': f'Error de base de datos: {str(e)}'}, 500
    except Exception as e:
        return {'error': f'Error inesperado: {str(e)}'}, 500


class DashboardHandler(http.server.SimpleHTTPRequestHandler):
    """Manejador personalizado para el dashboard"""
    
    def do_GET(self):
        parsed_path = urlparse(self.path)
        
        # Endpoint API para estadísticas
        if parsed_path.path == '/api/dashboard/stats':
            self.send_json_response(*get_dashboard_stats())
            return
            
        # Servir archivo estático (HTML, CSS, JS)
        super().do_GET()
    
    def send_json_response(self, data, status_code=200):
        """Envía respuesta JSON con headers CORS"""
        self.send_response(status_code)
        self.send_header('Content-type', 'application/json')
        self.send_header('Access-Control-Allow-Origin', '*')
        self.end_headers()
        self.wfile.write(json.dumps(data, ensure_ascii=False).encode('utf-8'))
    
    def log_message(self, format, *args):
        """Silencia logs innecesarios"""
        print(f"[Dashboard] {args[0]}")

        function actualizarDashboard() {
            fetch('/api/dashboard/stats')
                .then(response => {
                    if (!response.ok) throw new Error('HTTP ' + response.status);
                    return response.json();
                })
                .then(data => {
                    if (data.error) {
                        console.error('Error del servidor:', data.error);
                        document.getElementById('connection-status').className = 'status-badge status-error';
                        document.getElementById('connection-status').textContent = '● Error de conexión';
                        return;
                    }
                    
                    // Actualizar estado de conexión
                    document.getElementById('connection-status').className = 'status-badge status-ok';
                    document.getElementById('connection-status').textContent = '● Conectado';
                    
                    // Actualizar contadores con animación suave
                    animateValue('[data-canales]', data.canales_activos);
                    animateValue('[data-videos-totales]', data.videos_totales);
                    animateValue('[data-pendientes]', data.pendientes);
                    animateValue('[data-procesando]', data.procesando);
                    animateValue('[data-completados]', data.completados);
                    animateValue('[data-fallidos]', data.fallidos);
                    
                    // Actualizar timestamp
                    document.querySelector('[data-timestamp]').textContent = 
                        `Última actualización: ${data.ultima_actualizacion}`;
                    
                    // Actualizar actividad reciente
                    actualizarActividad(data.actividad_reciente);
                    
                    // Notificación si hay videos pendientes
                    if (data.pendientes > 0 && !window.pendientesNotificados) {
                        mostrarNotificacion(`📥 ${data.pendientes} videos nuevos pendientes de descargar`);
                        window.pendientesNotificados = true;
                    }
                    if (data.pendientes === 0) {
                        window.pendientesNotificados = false;
                    }
                })
                .catch(error => {
                    console.error('Error al actualizar dashboard:', error);
                    document.getElementById('connection-status').className = 'status-badge status-error';
                    document.getElementById('connection-status').textContent = '● Sin conexión';
                });
        }
        
        function animateValue(selector, newValue) {
            const element = document.querySelector(selector);
            if (!element) return;
            
            const currentValue = parseInt(element.textContent) || 0;
            if (currentValue === newValue) return;
            
            // Animación simple de conteo
            const duration = 300;
            const steps = 10;
            const increment = (newValue - currentValue) / steps;
            let step = 0;
            
            const timer = setInterval(() => {
                step++;
                const displayValue = Math.round(currentValue + increment * step);
                element.textContent = displayValue;
                
                if (step >= steps) {
                    element.textContent = newValue;
                    clearInterval(timer);
                }
            }, duration / steps);
        }
        
        function actualizarActividad(videos) {
            const container = document.querySelector('[data-actividad]');
            if (!videos || videos.length === 0) {
                container.innerHTML = '<div style="text-align:center;color:var(--text-secondary);padding:20px;">Sin actividad reciente</div>';
                return;
            }
            
            container.innerHTML = videos.map(video => {
                const fecha = video.date ? new Date(video.date).toLocaleString('es-ES', {
                    day: '2-digit', month: '2-digit', hour: '2-digit', minute: '2-digit'
                }) : 'N/A';
                
                return `
                    <div class="activity-item">
                        <span class="titulo" title="${video.title}">${video.title}</span>
                        <span class="fecha">${fecha}</span>
                        <span class="estado ${video.status || 'pending'}">${video.status || 'pendiente'}</span>
                    </div>
                `;
            }).join('');
        }
        
        function mostrarNotificacion(mensaje) {
            // Remover notificaciones anteriores
            const existing = document.querySelector('.toast');
            if (existing) existing.remove();
            
            const notif = document.createElement('div');
            notif.className = 'toast';
            notif.textContent = mensaje;
            document.body.appendChild(notif);
            
            setTimeout(() => {
                notif.style.animation = 'slideIn 0.3s ease reverse';
                setTimeout(() => notif.remove(), 300);
            }, 5000);
        }
        
        // ================= INICIALIZACIÓN =================
        
        // Ejecutar inmediatamente al cargar
        actualizarDashboard();
        
        // Actualizar cada 10 segundos
        setInterval(actualizarDashboard, 10000);
        
        // Permitir recarga manual con F5
        document.addEventListener('keydown', (e) => {
            if (e.key === 'F5') {
                e.preventDefault();
                actualizarDashboard();
            }
        });
        
        console.log('📊 Dashboard KDP Master cargado - Actualizando cada 10s');
    </script>

def main():
    """Inicia el servidor del dashboard"""
    # Crear archivo HTML temporal para servir
    html_path = BASE_DIR / "dashboard.html"
    
    try:
        with open(html_path, 'w', encoding='utf-8') as f:
            f.write(generate_dashboard_html())
        print(f"✅ Dashboard HTML generado: {html_path}")
    except Exception as e:
        print(f"⚠️ No se pudo generar HTML: {e}")
    
    # Configurar servidor
    with socketserver.TCPServer(("", PORT), DashboardHandler) as httpd:
        print(f"🚀 Dashboard disponible en: http://localhost:{PORT}")
        print(f"📁 Base de datos: {DB_PATH}")
        print("🔁 Auto-refresh: cada 10 segundos")
        print("ℹ️  Presiona Ctrl+C para detener")
        
        try:
            httpd.serve_forever()
        except KeyboardInterrupt:
            print("\n🛑 Servidor detenido")
            # Limpiar archivo temporal si existe
            if html_path.exists():
                html_path.unlink()


if __name__ == "__main__":
    main()

| Problema | Solución |

| Sin feedback de conexión | Badge de estado que cambia según respuesta del servidor |
